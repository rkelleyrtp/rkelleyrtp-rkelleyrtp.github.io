Static IP Address example

/etc/netplan/template.txt

# --------------------------
# Static IP Address Example
# --------------------------
network:
 version: 2
 renderer: networkd
 ethernets:
   eth0:
     dhcp4: no
     dhcp6: no
     addresses: [172.16.57.999/24]
     gateway4: 172.16.57.1
     nameservers:
       addresses: [172.16.57.5,172.16.57.8]

``

``

DHCP IP Address example

# --------------------------
# DHCP IP Address Example - Single interface
# --------------------------
network:
   version: 2
   renderer: networkd
   ethernets:
      eth0:
       dhcp4: yes

Trunk example

eth0 with static address eth1 with trunk/vlan interfaces

# --------------------------
# TRUNK interface example
# --------------------------
network:
    version: 2
    ethernets:
        eth0:
            addresses:
                - 172.16.155.91/24
            gateway4: 172.16.155.1
            nameservers:
                addresses: [172.16.155.254]
        eth1: {}

    vlans:
        vlan.101:
            id: 101
            link: eth1
            addresses: [10.20.30.91/24]
        vlan.102:
            id: 102
            link: eth1
            addresses: [40.50.60.91/24]

Bonding interfaces with static Address

# --------------------------
# BONDING example
# --------------------------
network:
version: 2
renderer: networkd
ethernets:
  eth0:
    dhcp4: false
    dhcp6: false
  eth1:
    dhcp4: false
    dhcp6: false

bonds:
  bond0:
    dhcp4: false
    dhcp6: false
    interfaces:
      - eth0
      - eth1
    addresses: [192.168.1.156/24]
    gateway4: 192.168.1.1
    parameters:
      mode: 802.3ad
    nameservers:
      addresses: [172.16.57.5,172.16.57.8,84.200.69.80,84.200.70.40]

Bonding interfaces with VLANs

# --------------------------
# BONDING example with VLANs
# --------------------------
network:
version: 2
renderer: networkd
ethernets:
  eth0:
    dhcp4: false
    dhcp6: false
  eth1:
    dhcp4: false
    dhcp6: false
bonds:
  bond0:
    dhcp4: false
    dhcp6: false
    interfaces:
      - eth0
      - eth1

    vlans:
        vlan.101:
            id: 101
            link: bond0
            addresses: [192.168.101.1/24]
        vlan.102:
            id: 102
            link: bond0
            addresses: [192.168.102.1/24]
